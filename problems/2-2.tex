\begin{problem}[问题2.2]
已知速度分布: $u=e^{-kt}x$, $v=-e^{-kt}y$. 求流线方程.
\end{problem}
\begin{solution}
\textbf{解:}由流线微分方程
\[
\frac{dx}{u} = \frac{dy}{v} \Longrightarrow \frac{dx}{e^{-kt}x} = \frac{dy}{-e^{-kt}y}
\]
积分得流线方程
\[
\int \frac{dx}{x} = \int \frac{dy}{-y} \Longrightarrow  xy = C
\]
其中$C$为常数.
\end{solution}
